IDENTIFIER      = (LETTER | "_") {LETTER | DIGIT | "_"} ;
INT_LITERAL     = NUMBER ;
FLOAT_LITERAL   = (NUMBER "." NUMBER) | ("." NUMBER) ;
STRING_LITERAL  = "\"" {ANY_CHAR} "\"" ;

ATOM = IDENTIFIER | INT_LITERAL | FLOAT_LITERAL | STRING_LITERAL ;

EXPR = (EXPR INF_OP EXPR) | (PRE_OP EXPR) | (EXPR POST_OP) | ATOM ;

RANGE = EXPR (".." | "..=") EXPR ;

IF_STATEMENT    = "if" EXPR BLOCK ;

FOR_STATEMENT = (FOR_IN_RANGE | FOR_C | FOR_WHILE) BLOCK ;
FOR_IN_RANGE  = FOR [IDENTIFIER "in"] RANGE ; 
FOR_C         = FOR [VARIABLE_DECLARATION]";" [EXPR]";"  [EXPR] ;
FOR_WHILE     = FOR EXPR;

RETURN_STATEMENT = "return" EXPR ;

DECLARATION = TOP_LVL_DECLARATION | SCOPED_DECLARATION ;

TOP_LVL_DECLARATION = FN_DECLARATION | EXPLICIT_VAR_DECLARATION | SHORT_VAR_DECLARATION | TYPE_DECLARATION ;

FN_DECLARATION = FN_SIGNATURE BLOCK ;
FN_SIGNATURE   = IDENTIFIER "::" "(" PARAMS_OR_RETURN_TYPES_LIST ")" [ "->" TYPE |  ( "(" PARAMS_OR_RETURN_TYPES_LIST ")" ) ] ; 
PARAMS_OR_RETURN_TYPES_LIST = (TYPE {"," TYPE}) | (IDENTIFIER TYPE {"," IDENTIFIER TYPE}) ;


STATEMENT = (IF_STATEMENT | FOR_STATEMENT | RETURN_STATEMENT | DECLARATION) ;
BLOCK = "{" { STATEMENT } "}"


NUMBER = DIGIT {DIGIT | ("_" DIGIT)};
LETTER = ANY_LETTER;
DIGIT  = 0-9;
