IDENTIFIER      = (LETTER | "_") {LETTER | DIGIT | "_"} ;
INT_LITERAL     = NUMBER ;
FLOAT_LITERAL   = (NUMBER "." NUMBER) | ("." NUMBER) ;
STRING_LITERAL  = "\"" {ANY_CHAR} "\"" ;

ATOM = IDENTIFIER | INT_LITERAL | FLOAT_LITERAL | STRING_LITERAL ;

EXPR = (EXPR INF_OP EXPR) | (PRE_OP EXPR) | (EXPR POST_OP) | ATOM ;

INF_OP  = "." | "::" | "+" | "-" | "*" | "/" | "%" | "&&" | "||" | "==" | "!=" | "<" | "<=" | ">" | ">=" | "|" | "&" | "<<" | ">>" | "^" ; 
PRE_OP  = "+" | "-" | "&" | "*" | "!" | "~" ; 
POST_OP = "++" | "--" ;


STATEMENT = (IF_STATEMENT | FOR_STATEMENT | RETURN_STATEMENT | DECLARATION | EXPR) ;

DECLARATION = FN_DECLARATION | VARIABLE_DECLARATION | TYPE_DECLARATION ;

FN_DECLARATION = FN_SIGNATURE BLOCK ;
FN_SIGNATURE   = IDENTIFIER "::" "(" [ PARAMS_OR_RETURN_TYPES_LIST ] ")" [ "->" TYPE |  ( "(" PARAMS_OR_RETURN_TYPES_LIST ")" ) ] ; 
PARAMS_OR_RETURN_TYPES_LIST = (TYPE {"," TYPE}) | (IDENTIFIER TYPE {"," IDENTIFIER TYPE}) ;

VARIABLE_DECLARATION = EXPLICIT_VAR_DECLARATION | SHORT_VAR_DECLARATION ;

EXPLICIT_VAR_DECLARATION = "let" IDENTIFIER (([":" TYPE] "=" EXPR) | (":" TYPE)) ; 
SHORT_VAR_DECLARATION    = IDENTIFIER ":=" EXPR ; 

TYPE_DECLARATION   = TYPE_ALIAS | STRUCT_DECLARATION | ENUM_DECLARATION | FN_SIGNATURE ;
TYPE_ALIAS         = IDENTIFIER "::" (IDENTIFIER | TYPE) ;
STRUCT_DECLARATION = IDENTIFIER "::" "struct" "{" [IDENTIFIER ":" TYPE {"," IDENTIFIER ":" TYPE}] "}" ; 
ENUM_DECLARATION   = IDENTIFIER "::" "enum" "{" IDENTIFIER {"," IDENTIFIER} "}"

IF_STATEMENT = "if" EXPR BLOCK ;

RANGE = EXPR (".." | "..=") EXPR ;

FOR_STATEMENT = (FOR_IN_RANGE | FOR_C | FOR_WHILE) BLOCK ;
FOR_IN_RANGE  = "for" IDENTIFIER "in" RANGE ; 
FOR_C         = "for" [VARIABLE_DECLARATION]";" [EXPR]";"  [EXPR] ;
FOR_WHILE     = "for" EXPR ;

RETURN_STATEMENT = "return" EXPR ;
BLOCK = "{" { STATEMENT } "}" ; 


NUMBER = DIGIT {DIGIT | ("_" DIGIT)} ;
LETTER = ANY_LETTER ;
DIGIT  = 0-9 ;
